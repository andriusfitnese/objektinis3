182 void shrink_to_fit() {
183 if (size_ == capacity_)
184 return;
185
186 // 1) remember the current size
187 size_type old_size = size_;
188
189 // 2) allocate exactly old_size slots
190 pointer new_data = std::allocator_traits<Allocator>::
191 allocate(allocator_, old_size);
192
193 // 3) move-construct each element into new_data
194 size_type i = 0;
195 try {
196 for (; i < old_size; ++i) {
197 std::allocator_traits<Allocator>::construct(
198 allocator_,
199 new_data + i,
200 std::move_if_noexcept(data_[i])
201 );
202 }
203 }
204 catch (...) {
205 // if any construct throws, roll back what we built so far
206 for (size_type j = 0; j < i; ++j) {
207 std::allocator_traits<Allocator>::destroy(
208 allocator_, new_data + j
209 );
210 }
211 std::allocator_traits<Allocator>::deallocate(
212 allocator_, new_data, old_size
213 );
214 throw;
215 }
216
217 // 4) destroy old elements (but do not touch size_ yet)
218 for (size_type j = 0; j < old_size; ++j) {
219 std::allocator_traits<Allocator>::destroy(
220 allocator_, data_ + j
221 );
222 }
223 // 5) free old storage
224 std::allocator_traits<Allocator>::deallocate(
225 allocator_, data_, capacity_
226 );
227
228 // 6) commit the new buffer
229 data_ = new_data;
230 capacity_ = old_size;
231 size_ = old_size; // restore size
232 }

541 // Read initial [
542 char ch;
543 if (!(is >> ch) || ch != '[') {
544 is.setstate(std::ios::failbit);

546 }
547
548 // Now read values until ']' or failure
